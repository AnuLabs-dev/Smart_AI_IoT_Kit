ESP32-CAM & Arduino IDE Setup

ESP32-CAM features, GPIO pin reference, camera specifications and step-by-step Arduino IDE installation guide.

⚡ ESP32-CAM Key Features
📷

OV2640 Camera

2MP camera supporting UXGA (1600×1200) resolution, JPEG/BMP output.

📶

Wi-Fi + Bluetooth

802.11 b/g/n 2.4GHz Wi-Fi and Bluetooth 4.2 + BLE on one chip.

🧠

Dual-Core 240 MHz

Xtensa LX6 — handles video streaming and IoT control simultaneously.

💾

4MB PSRAM

Extra RAM for frame buffers required by camera and face-detection models.

🗂️

MicroSD Slot

Store captured images and video locally on up to 4GB MicroSD card.

🔦

Onboard Flash LED

White LED for low-light photography, controlled via GPIO 4.

⚠️
The ESP32-CAM does not have a built-in USB port. You must use the included Programming Cable (USB-to-Serial adapter) to upload code. Connect: TX→RX0, RX→TX0, GND→GND, 5V→5V, IO0→GND (for flash mode).

🚫
GPIOs 34, 35, 36, 39 are input-only — they cannot be used as outputs. GPIO 6–11 are connected internally to SPI flash and must never be used.
💻 Arduino IDE Setup
1
Download and install Arduino IDE 2.x from arduino.cc/en/software
2
File → Preferences → paste this URL in Additional boards manager URLs:
https://raw.githubusercontent.com/espressif/arduino-esp32/gh-pages/package_esp32_index.json
3
Tools → Board → Boards Manager → search esp32 → install ESP32 by Espressif Systems v2.x
4
Tools → Board → ESP32 Arduino → select AI Thinker ESP32-CAM
5
Connect your programming cable. Bridge IO0 to GND. Select port under Tools → Port.
6
Click Upload ▶. When "Connecting..." appears, press the RESET button on the board briefly.
7
After upload: remove the IO0-to-GND bridge, press RESET again to run the sketch.
💡
Install required libraries via Sketch → Include Library → Manage Libraries: DHT sensor library, Adafruit BMP280, Adafruit SSD1306, ESP32Servo, MFRC522 (RFID), Stepper.

Sensors Complete Guide

🌡️ DHT11 Temperature & Humidity Sensor
DHT11 PinESP32-CAM
VCC3.3V
DATAGPIO 13
GNDGND
💡
Place a 10kΩ pull-up resistor between VCC and DATA pin for reliable communication.
📦
Requires: DHT sensor library by Adafruit — install from Library Manager.
C++dht11_sensor.ino
#include <DHT.h>
#define DHTPIN 13
#define DHTTYPE DHT11
DHT dht(DHTPIN, DHTTYPE);

void setup() {
  Serial.begin(115200);
  dht.begin();
}

void loop() {
  float h = dht.readHumidity();
  float t = dht.readTemperature();
  if(isnan(h) || isnan(t)) {
    Serial.println("Read failed!");
    return;
  }
  Serial.printf("Temp: %.1f°C  Hum: %.1f%%\n", t, h);
  delay(2000);
}
🌬️ BMP280 Pressure & Temperature Sensor (I2C)
BMP280 PinESP32-CAM
VCC3.3V
GNDGND
SCLGPIO 14 (SCL)
SDAGPIO 15 (SDA)
📡
Default I2C address: 0x76 (or 0x77 if SDO pin is HIGH).
C++bmp280.ino
#include <Wire.h>
#include <Adafruit_BMP280.h>
Adafruit_BMP280 bmp;

void setup() {
  Serial.begin(115200);
  Wire.begin(15, 14); // SDA, SCL
  if(!bmp.begin(0x76)) {
    Serial.println("BMP280 not found!");
    while(1);
  }
}

void loop() {
  Serial.printf("Temp:%.1fC  Press:%.1fhPa  Alt:%.1fm\n",
    bmp.readTemperature(),
    bmp.readPressure()/100.0F,
    bmp.readAltitude(1013.25));
  delay(2000);
}
📡 HC-SR04 Ultrasonic Distance Sensor
HC-SR04 PinESP32-CAM
VCC5V
TRIGGPIO 12
ECHOGPIO 13
GNDGND
⚠️
ECHO pin outputs 5V. Use a 1kΩ + 2kΩ voltage divider to reduce to 3.3V to protect ESP32-CAM.
C++ultrasonic.ino
#define TRIG 12
#define ECHO 13

void setup() {
  Serial.begin(115200);
  pinMode(TRIG, OUTPUT);
  pinMode(ECHO, INPUT);
}

void loop() {
  digitalWrite(TRIG, LOW); delayMicroseconds(2);
  digitalWrite(TRIG, HIGH); delayMicroseconds(10);
  digitalWrite(TRIG, LOW);
  long dur = pulseIn(ECHO, HIGH);
  float dist = dur * 0.034 / 2;
  Serial.printf("Distance: %.1f cm\n", dist);
  delay(500);
}

🦾 SG90 Servo Motor
Servo WireESP32-CAM
Red (VCC)5V
Brown (GND)GND
Orange (Signal)GPIO 12

Range: 0° to 180°. PWM signal: 50Hz, 1ms (0°) to 2ms (180°) pulse width.

📦
Requires: ESP32Servo library — install via Library Manager.
C++servo_sweep.ino
#include <ESP32Servo.h>
Servo myServo;

void setup() {
  myServo.attach(12);
}

void loop() {
  // Sweep 0° to 180°
  for(int pos=0; pos<=180; pos++) {
    myServo.write(pos);
    delay(15);
  }
  // Sweep back 180° to 0°
  for(int pos=180; pos>=0; pos--) {
    myServo.write(pos);
    delay(15);
  }
}

  if(!rfid.PICC_ReadCardSerial())  return;

  for(byte i=0; i<rfid.uid.size; i++)
    Serial.printf("%02X ", rfid.uid.uidByte[i]);
  Serial.println();
}
🔘 Push Button with Debounce
Button PinESP32-CAMNotes
Pin 1GPIO 13Signal pin
Pin 2GNDOther side of button
C++button_debounce.ino
#define BTN 13
bool lastState = HIGH, ledState = false;
unsigned long lastDebounce = 0;

void setup() {
  pinMode(BTN, INPUT_PULLUP);  // Use internal pull-up
  pinMode(4, OUTPUT);
}

void loop() {
  bool reading = digitalRead(BTN);
  if(reading != lastState) lastDebounce = millis();
  if(millis() - lastDebounce > 50) { // 50ms debounce
    if(reading == LOW) {  // Button pressed
      ledState = !ledState;
      digitalWrite(4, ledState);
    }
  }
  lastState = reading;
}

RFID Access Control System

RFID Relay Buzzer
ComponentESP32-CAM
RC522 SDAGPIO 13
RC522 SCKGPIO 14
Relay SignalGPIO 12
BuzzerGPIO 15
1
Scan your card and note the UID from Serial Monitor
2
Add the UID to the authorizedUIDs array in the sketch
3
Authorized card → relay activates (door unlocks) + green LED
C++rfid_access.ino
#include <SPI.h>
#include <MFRC522.h>
MFRC522 rfid(13, 15); // SS, RST
String authorizedUID = "AB 12 34 CD";
#define RELAY 12

void loop() {
  if(!rfid.PICC_IsNewCardPresent()) return;
  if(!rfid.PICC_ReadCardSerial())  return;
  String uid = "";
  for(byte i=0; i<rfid.uid.size; i++)
    uid += String(rfid.uid.uidByte[i], HEX)+" ";
  uid.trim(); uid.toUpperCase();
  if(uid == authorizedUID) {
    Serial.println("✅ ACCESS GRANTED");
    digitalWrite(RELAY, LOW);  // Unlock
    delay(3000);
    digitalWrite(RELAY, HIGH); // Re-lock
  } else Serial.println("❌ ACCESS DENIED");
}

#define GAS_A  34

      digitalWrite(LED, LOW);

  }

}

🔍 AI Project 2 — Object Detection with TensorFlow Lite
📦
Requires installing TensorFlowLite_ESP32 library from Library Manager. Uses a quantized MobileNet model (person detection) compiled for ESP32.
C++person_detection.ino
#include <TensorFlowLite_ESP32.h>
#include "person_detect_model_data.h"
#include "esp_camera.h"

// Setup TFLite interpreter, model, and tensor arena
const int kTensorArenaSize = 70 * 1024;
uint8_t tensor_arena[kTensorArenaSize];

void loop() {
  camera_fb_t* fb = esp_camera_fb_get();
  if(!fb) return;
  
  // Run inference on captured frame
  TfLiteStatus invoke_status = interpreter->Invoke();
  
  // Read output: person score vs no-person score
  int8_t* output = interpreter->output(0)->data.int8;
  bool person = output[1] > output[0];
  
  Serial.println(person ? "👤 Person detected!"
                        : "No person in frame");
  esp_camera_fb_return(fb);
}
⚠️
TFLite inference uses significant RAM. Disable Wi-Fi during inference if you encounter memory errors: WiFi.mode(WIFI_OFF) before running the model.

FOMO Object Detection — Count & Locate Objects

Edge Impulse FOMO Bounding Boxes

FOMO (Faster Objects More Objects) is Edge Impulse's ultra-lightweight object detection model designed specifically for microcontrollers. It can detect and count multiple objects in a frame at ~30 FPS on ESP32-CAM using only ~50KB RAM.

Edge Impulse SettingsValue
Input blockImage (96×96, RGB or Grayscale)
Processing blockImage
Learning blockObject Detection (FOMO)
ArchitectureFOMO MobileNetV2 0.35
Labeling methodBounding boxes (draw in browser)
Min samples/class60+ images recommended
⚠️
FOMO outputs object centroids, not full bounding boxes. It excels at counting and locating small, distinct objects (screws, coins, pills) but is not designed for overlapping objects.
✅
Great use cases: counting components on a workbench, detecting presence of a specific item, parking spot occupancy, bottle-cap inspection.
C++ei_fomo_detect.ino
#include <your-project-name_inferencing.h>
#include "esp_camera.h"
#include "edge-impulse-sdk/dsp/image/image.hpp"

// (camera init same as previous project)

void loop() {
  camera_fb_t* fb = esp_camera_fb_get();
  if(!fb) return;

  // Convert JPEG → RGB888 → resize to 96×96
  uint8_t* buf = new uint8_t[96*96*3];
  jpg2rgb888(fb->buf, fb->len, buf, JPG_SCALE_NONE);
  esp_camera_fb_return(fb);

  ei::signal_t signal;
  ei_camera_cutout_get_data(buf, &signal);

  ei_impulse_result_t result;
  run_classifier(&signal, &result, false);
  delete[] buf;

  // Iterate bounding box results
  uint32_t count = 0;
  for(auto bb : result.bounding_boxes) {
    if(bb.value < 0.5f) continue; // confidence threshold
    count++;
    Serial.printf("  %s (%.0f%%) @ x:%d y:%d w:%d h:%d\n",
      bb.label, bb.value*100,
      bb.x, bb.y, bb.width, bb.height);
  }
  Serial.printf("Total detected: %u\n\n", count);
  delay(500);
}

#include <your-ei-project_inferencing.h>
#include "esp_camera.h"
#include <WiFi.h>
#include <HTTPClient.h>
#define PIR    13
#define RELAY  12
#define BUZZER 15
const char* TG_TOKEN = "YOUR_BOT_TOKEN";
const char* TG_CHAT  = "YOUR_CHAT_ID";

void sendTelegramAlert() {
  HTTPClient http;
  String url = "https://api.telegram.org/bot"
    + String(TG_TOKEN) + "/sendMessage?chat_id="
    + TG_CHAT + "&text=🚨+Intruder+Detected!";
  http.begin(url);
  http.GET();
  http.end();
}

void loop() {
  if(digitalRead(PIR) != HIGH) return;
  Serial.println("PIR triggered — running AI...");

  camera_fb_t* fb = esp_camera_fb_get();
  if(!fb) return;

  // Run FOMO inference (populate signal first)
  ei_impulse_result_t result;
  // ... (signal setup as in Project 6)
  run_classifier(&signal, &result, false);

  bool person_found = false;
  for(auto bb : result.bounding_boxes) {
    if(strcmp(bb.label, "person")==0
       && bb.value > 0.6f) {
      person_found = true; break;
    }
  }
  esp_camera_fb_return(fb);

  if(person_found) {
    digitalWrite(RELAY,  LOW);
    digitalWrite(BUZZER, HIGH);
    sendTelegramAlert();
    delay(5000);
    digitalWrite(RELAY,  HIGH);
    digitalWrite(BUZZER, LOW);
    delay(30000); // cooldown
  }
}

🔶 4-Band Formula

B1 = 1st digit | B2 = 2nd digit | B3 = Multiplier | B4 = Tolerance

Value = (B1×10 + B2) × Multiplier

Example: Red-Red-Brown-Gold = (22)×10 = 220Ω ±5%

🔷 5-Band Formula (Precision)

B1-B2-B3 = digits | B4 = Multiplier | B5 = Tolerance

Value = (B1×100 + B2×10 + B3) × Multiplier

Example: Brn-Blk-Blk-Org-Brn = 100×1K = 100KΩ ±1%

🥇
Reading tip: Always hold the resistor with the Gold/Silver tolerance band on the RIGHT, then read left to right. Memory trick: B.B. ROY of Great Britain Very Good Wife → Black, Brown, Red, Orange, Yellow, Green, Blue, Violet, Grey, White (0–9).
